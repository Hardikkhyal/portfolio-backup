## 1. Welcome.jsx (Client Love Section)
*Path: `src/components/Welcome.jsx`*
*Description: Contains the screen welcome intro typography and the "Client Love" infinite scrolling text marquees using custom React-rendered cards.*

```jsx
import React, { useEffect, useRef } from 'react';
import gsap from 'gsap';
import { ScrollTrigger } from 'gsap/ScrollTrigger';
import { Star } from 'lucide-react';

// Register GSAP ScrollTrigger plugin
gsap.registerPlugin(ScrollTrigger);

export default function Welcome({ testimonials }) {
  const welcomeRef = useRef(null);
  const welcomeTextRef = useRef(null);

  useEffect(() => {
    // GSAP context scope to handle clean garbage collection on component unmount
    const ctx = gsap.context(() => {
      // Parallax text shift for the welcome screen typography
      gsap.fromTo(welcomeTextRef.current,
        { y: 100 },
        {
          y: -100,
          scrollTrigger: {
            trigger: welcomeRef.current,
            start: 'top bottom',
            end: 'bottom top',
            scrub: true,
          }
        }
      );
    }, welcomeRef);

    return () => ctx.revert();
  }, []);

  // Split testimonials for alternating speed/direction row lists
  const midPoint = Math.ceil(testimonials.length / 2);
  const row1Testimonials = [...testimonials.slice(0, midPoint), ...testimonials.slice(0, midPoint), ...testimonials.slice(0, midPoint)];
  const row2Testimonials = [...testimonials.slice(midPoint), ...testimonials.slice(midPoint), ...testimonials.slice(midPoint)];

  return (
    <div 
      ref={welcomeRef} 
      id="about"
      // uses the 'noise-overlay' CSS class defined in the global styles below
      className="relative min-h-[200vh] w-full flex flex-col justify-between overflow-hidden bg-gradient-to-b from-white to-zinc-50 select-none noise-overlay text-zinc-900"
    >
      {/* Screen 1: Welcome Typography Reveal (Light Theme) */}
      <div className="relative h-[100vh] w-full flex flex-col justify-center items-center px-6 z-10">
        <div ref={welcomeTextRef} className="flex flex-col items-center">
          <span className="text-zinc-400 uppercase tracking-widest text-xs md:text-sm font-semibold mb-4">
            Creative Space // 2026
          </span>
          {/* uses 'text-stroke-dark' helper CSS for thin-border transparent typography */}
          <h2 className="text-[20vw] md:text-[25vw] font-black leading-none tracking-tight uppercase text-stroke-dark text-zinc-900/[0.03] select-none">
            WELCOME
          </h2>
          <p className="text-zinc-500 text-sm md:text-xl uppercase tracking-widest -mt-6 md:-mt-12 font-medium z-10">
            To my creative digital space
          </p>
        </div>
      </div>

      {/* Screen 2: Testimonials Marquee Section (Light Theme) */}
      <div className="relative h-[100vh] w-full flex flex-col justify-center items-center px-4 md:px-12 py-20 z-10 bg-white border-t border-zinc-200/50">
        <div className="max-w-4xl text-center mb-12" data-aos="fade-up">
          <span className="text-zinc-400 font-bold uppercase tracking-widest text-xs md:text-sm mb-2 block">
            // Client Reviews
          </span>
          <h3 className="text-4xl md:text-6xl font-black uppercase tracking-tight text-zinc-900 mb-4">
            Client Love
          </h3>
          <p className="text-zinc-500 font-medium text-xs md:text-base max-w-lg mx-auto">
            Explore what clients and designers around the world say about crafting products with Lema.web.
          </p>
        </div>

        {/* Testimonials Infinite Marquees Container */}
        <div className="w-full flex flex-col gap-6 overflow-hidden py-4">
          
          {/* Row 1: Left-moving Marquee (uses .animate-marquee-left CSS styling) */}
          <div className="relative w-full flex overflow-x-hidden">
            <div className="animate-marquee-left flex gap-6 py-2">
              {row1Testimonials.map((review, idx) => (
                <div 
                  key={`row1-${idx}-${review.name}`}
                  className="w-[300px] md:w-[380px] flex-shrink-0 p-6 rounded-2xl bg-white border border-zinc-200/80 hover:border-brand-yellow/60 shadow-sm hover:shadow-xl transition-all duration-300 hover:scale-102 flex flex-col justify-between"
```

## 4. Smooth Scroll (Lenis & GSAP) Context
*Path: `src/App.jsx`*
*Description: Shows how the Lenis smooth scrolling driver is configured and synchronized with the GSAP ScrollTrigger rendering tick system.*

```javascript
// Add inside the App component useEffect block
useEffect(() => {
  // Initialize Lenis smooth scroll
  const lenis = new Lenis({
    duration: 1.2,
    smoothWheel: true,
    smoothTouch: true,
    touchMultiplier: 2,
  });

  // Connect Lenis scroll callback to GSAP ScrollTrigger updates
  lenis.on('scroll', ScrollTrigger.update);

  // Coordinate GSAP ticker with Lenis frame cycles
  gsap.ticker.add((time) => {
    lenis.raf(time * 1000);
  });

  // Disable GSAP lag smoothing to maintain lock-step synchronization
  gsap.ticker.lagSmoothing(0);

  // Animation frame loop
  const raf = (time) => {
    lenis.raf(time);
    requestAnimationFrame(raf);
  };
  requestAnimationFrame(raf);

  // Cleanup handler
  return () => {
    lenis.destroy();
    ScrollTrigger.getAll().forEach(trigger => trigger.kill());
  };
}, []);
```
